<script>
// ─── STATE ────────────────────────────────────────────────────────────────────
let tabs = [];
let activeTabId = null;
let tabCounter = 0;
let wordWrap = true;
let showLineNums = true;
let showStatusBar = true;
let showToolbarFlag = true;
let darkMode = false;
let currentFontSize = 14;
let findMatches = [];
let findIndex = -1;

// ─── TAB MANAGEMENT ───────────────────────────────────────────────────────────
function createTab(name = 'Untitled', content = '') {
  const id = ++tabCounter;
  tabs.push({ id, name, content, modified: false, encoding: 'UTF-8', eol: 'CRLF', savedContent: '' });
  return id;
}

function newTab(content = '', name = 'Untitled') {
  const id = createTab(name, content);
  renderTabs();
  switchTab(id);
}

function switchTab(id) {
  if (activeTabId !== null) {
    const current = tabs.find(t => t.id === activeTabId);
    if (current) current.content = editor.value;
  }
  activeTabId = id;
  const tab = tabs.find(t => t.id === id);
  if (!tab) return;
  editor.value = tab.content;
  updateLineNumbers();
  updateStatus();
  renderTabs();
  document.title = (tab.modified ? '● ' : '') + tab.name + ' — Notepad';
}

function closeTab(id) {
  const idx = tabs.findIndex(t => t.id === id);
  if (idx === -1) return;
  const tab = tabs[idx];
  if (tab.modified) {
    if (!confirm(`Save changes to "${tab.name}" before closing?`)) {
      // still close
    }
  }
  tabs.splice(idx, 1);
  if (tabs.length === 0) { newTab(); return; }
  if (activeTabId === id) {
    const nextIdx = Math.min(idx, tabs.length - 1);
    switchTab(tabs[nextIdx].id);
  } else {
    renderTabs();
  }
}

function renderTabs() {
  const container = document.getElementById('tabs');
  container.innerHTML = '';
  tabs.forEach(tab => {
    const el = document.createElement('div');
    el.className = 'tab' + (tab.id === activeTabId ? ' active' : '');
    el.innerHTML = `<span class="tab-name">${escHtml(tab.name)}</span>${tab.modified ? '<span class="tab-modified">●</span>' : ''}<button class="tab-close" onclick="event.stopPropagation();closeTab(${tab.id})" title="Close">×</button>`;
    el.onclick = () => switchTab(tab.id);
    container.appendChild(el);
  });
}

function markModified() {
  const tab = tabs.find(t => t.id === activeTabId);
  if (tab && !tab.modified) {
    tab.modified = true;

function editorKeydown(e) {
  const isMac = navigator.platform.includes('Mac');
  const ctrl = isMac ? e.metaKey : e.ctrlKey;

  if (e.key === 'Tab' && !e.shiftKey) {
    e.preventDefault();
    insertAtCursor('    ');
    return;
  }
  if (e.key === 'Tab' && e.shiftKey) {
    e.preventDefault();
    // dedent
    const start = editor.selectionStart;
    const before = editor.value.substring(0, start);
    const lineStart = before.lastIndexOf('\n') + 1;
    const lineContent = editor.value.substring(lineStart);
    if (lineContent.startsWith('    ')) {
      editor.value = editor.value.substring(0, lineStart) + lineContent.substring(4);
      editor.selectionStart = editor.selectionEnd = Math.max(lineStart, start - 4);
    }
    return;
  }
  if (e.key === 'F5') { e.preventDefault(); insertDateTime(); }
  if (e.key === 'F1') { e.preventDefault(); showHelp(); }
  if (ctrl && e.key === 'f') { e.preventDefault(); openFind(); }
  if (ctrl && e.key === 'h') { e.preventDefault(); openReplace(); }
  if (ctrl && e.key === 'g') { e.preventDefault(); goToLine(); }
  if (ctrl && e.key === 'n') { e.preventDefault(); newTab(); }
  if (ctrl && e.key === 'w') { e.preventDefault(); closeTab(activeTabId); }
  if (ctrl && e.key === 's') { e.preventDefault(); if (e.shiftKey) saveAs(); else saveFile(); }
  if (ctrl && e.key === 'o') { e.preventDefault(); openFile(); }
  if (ctrl && e.key === 'p') { e.preventDefault(); printFile(); }
  if (ctrl && e.key === '=') { e.preventDefault(); zoomIn(); }
  if (ctrl && e.key === '-') { e.preventDefault(); zoomOut(); }
  if (ctrl && e.key === '0') { e.preventDefault(); zoomReset(); }
  if (ctrl && e.shiftKey && e.key === 'U') { e.preventDefault(); transformUpper(); }
  if (ctrl && e.shiftKey && e.key === 'L') { e.preventDefault(); transformLower(); }

  // Auto-pair brackets
  const pairs = { '(': ')', '[': ']', '{': '}', '"': '"', "'": "'" };
  if (pairs[e.key] && !e.ctrlKey && !e.metaKey) {
    const sel = editor.value.substring(editor.selectionStart, editor.selectionEnd);
    if (sel.length > 0) {
      e.preventDefault();
      const wrapped = e.key + sel + pairs[e.key];
      insertAtCursor(wrapped);
      return;
    }
  }

  // Enter: auto-indent
  if (e.key === 'Enter') {
    const pos = editor.selectionStart;
    const before = editor.value.substring(0, pos);
    const lastLine = before.split('\n').pop();
    const indent = lastLine.match(/^(\s*)/)[1];
    if (indent) {
      e.preventDefault();
      insertAtCursor('\n' + indent);
    }
  }
}

function insertAtCursor(text) {
  const start = editor.selectionStart;
  const end = editor.selectionEnd;
  editor.value = editor.value.substring(0, start) + text + editor.value.substring(end);
  editor.selectionStart = editor.selectionEnd = start + text.length;
  editor.dispatchEvent(new Event('input'));
}

function saveFile() {

  a.href = url; a.download = tab.name; a.click();
  URL.revokeObjectURL(url);
  tab.modified = false;
  tab.savedContent = editor.value;
  renderTabs();
  document.title = tab.name + ' — Notepad';
}

function closeFindbar() {
  document.getElementById('findbar').classList.remove('open');
  findMatches = []; findIndex = -1;
  document.getElementById('find-count').textContent = '';
  editor.focus();
}

function updateFindHighlights() {
  const query = document.getElementById('find-input').value;
  const caseSens = document.getElementById('find-case').checked;
  const whole = document.getElementById('find-whole').checked;
  findMatches = [];
  document.getElementById('find-count').textContent = '';
  if (!query) return;
  const text = editor.value;
  const flags = caseSens ? 'g' : 'gi';
  const pattern = whole ? `\\b${escRegex(query)}\\b` : escRegex(query);
  let re;
  try { re = new RegExp(pattern, flags); } catch { return; }
  let m;
  while ((m = re.exec(text)) !== null) findMatches.push(m.index);
  document.getElementById('find-count').textContent =
    findMatches.length ? `${Math.min(findIndex + 1, findMatches.length)} / ${findMatches.length}` : 'Not found';
}

function findNext() {
  if (!findMatches.length) return;
  findIndex = (findIndex + 1) % findMatches.length;
  selectMatch();
}

function findPrev() {
  if (!findMatches.length) return;
  findIndex = (findIndex - 1 + findMatches.length) % findMatches.length;
  selectMatch();
}

function selectMatch() {
  const query = document.getElementById('find-input').value;
  editor.focus();
  editor.selectionStart = findMatches[findIndex];
  editor.selectionEnd = findMatches[findIndex] + query.length;
  document.getElementById('find-count').textContent = `${findIndex + 1} / ${findMatches.length}`;
}

function findKeydown(e) {
  if (e.key === 'Enter') { e.shiftKey ? findPrev() : findNext(); }
  if (e.key === 'Escape') closeFindbar();
}

function replaceOne() {
  if (findIndex < 0 || !findMatches.length) { findNext(); return; }
  const query = document.getElementById('find-input').value;
  const rep = document.getElementById('replace-input').value;
  const pos = findMatches[findIndex];
  editor.value = editor.value.substring(0, pos) + rep + editor.value.substring(pos + query.length);
  editor.dispatchEvent(new Event('input'));
  updateFindHighlights();
  findNext();
}

function replaceAll() {
  const query = document.getElementById('find-input').value;
  const rep = document.getElementById('replace-input').value;
  const caseSens = document.getElementById('find-case').checked;
  const whole = document.getElementById('find-whole').checked;
  if (!query) return;
  const flags = caseSens ? 'g' : 'gi';
  const pattern = whole ? `\\b${escRegex(query)}\\b` : escRegex(query);
  let re;
  try { re = new RegExp(pattern, flags); } catch { return; }
  const count = (editor.value.match(re) || []).length;
  editor.value = editor.value.replace(re, rep);
  editor.dispatchEvent(new Event('input'));
  updateFindHighlights();
  if (count) alert(`Replaced ${count} occurrence(s).`);
}

function transformUpper() {
  const s = editor.selectionStart, e = editor.selectionEnd;
  if (s === e) return;
  editor.value = editor.value.substring(0, s) + editor.value.substring(s, e).toUpperCase() + editor.value.substring(e);
  editor.selectionStart = s; editor.selectionEnd = e;
  markModified();
}

function transformLower() {
  const s = editor.selectionStart, e = editor.selectionEnd;
  if (s === e) return;
  editor.value = editor.value.substring(0, s) + editor.value.substring(s, e).toLowerCase() + editor.value.substring(e);
  editor.selectionStart = s; editor.selectionEnd = e;
  markModified();
}

function transformBold() { insertAtCursor('**text**'); }

function trimLines() {
  editor.value = editor.value.split('\n').map(l => l.trimEnd()).join('\n');
  editor.dispatchEvent(new Event('input'));
}

function sortLines() {
  editor.value = editor.value.split('\n').sort((a, b) => a.localeCompare(b)).join('\n');
  editor.dispatchEvent(new Event('input'));
}

function removeDuplicates() {
  const lines = editor.value.split('\n');
  const seen = new Set();
  editor.value = lines.filter(l => { if (seen.has(l)) return false; seen.add(l); return true; }).join('\n');
  editor.dispatchEvent(new Event('input'));
}
